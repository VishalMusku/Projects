#Group the data so it can be counted
breaches_per_year = df.groupby(['year','Type_of_Breach'])['Type_of_Breach'].count()

#Most frequent
max_breach = {}
for year in df['year'].unique():
    max_breach[year] = breaches_per_year[year].idxmax()    

#Output per year

breachTypeList=list(max_breach.items())
top_breachesList=list(max_breach.values())

breachTypeList.sort()
print('\nRESULT:\n')

for item in breachTypeList:
    print('Year', item[0], 'had', item[1], 'with', breaches_per_year[item[0]][max_breach[item[0]]], 'breaches.')

# create a dictionary to map types to colors
colors = {'Hacking/IT Incident': 'blue',
          'Unauthorized Access/Disclosure': 'orange',
          'Improper Disposal': 'green',
          'Theft': 'red',
          'Loss, Unauthorized Access/Disclosure': 'purple'}

# create a new figure with a larger size
plt.figure(figsize=(10, 6))

# group the data by type
grouped_maxitems = maxitems.groupby('Type_of_Breach')

# create a bar plot for each type
for name, group in grouped_maxitems:
    plt.bar(group['Year'], group['Counts'], color=colors[name], label=name)

# set the x-axis ticks and labels
plt.xticks(range(maxitems['Year'].min(), maxitems['Year'].max()+1))

# add legend, axis labels, and title
plt.legend()
plt.xlabel('Year')
plt.ylabel('Count')
plt.title('Count by Year and Type')

# show the plot
plt.show()

main_dict={}
for x in df['Type_of_Breach'].unique():
    data = df[df['Type_of_Breach'] == x].sort_values(by='Location_of_Breached_Information', ascending=False)
    data_grouped = data.groupby(['Type_of_Breach', 'Location_of_Breached_Information'])['Location_of_Breached_Information'].size().reset_index(name='counts')
    top3 = data_grouped.sort_values(by=['counts'], ascending=False).iloc[:3]
    temp_dict={}
    if not top3.empty:
        print('\nTOP 3 values for ', "\'",top3['Type_of_Breach'].iloc[0],"\' ", 'are: \n')
        i=1
        for index, row in top3.iterrows():
            i
            print(i,".",row['Location_of_Breached_Information'], " : ", row['counts'], 'times.')
            i=i+1
            temp_dict.update({row['Location_of_Breached_Information']:row['counts']})
     
    main_dict.update({x:temp_dict})
